# Water Jug Problem using Production Rules

from collections import deque

# Jug capacities
cap1 = 4
cap2 = 3

# Target
target = 2

visited = set()

def water_jug():
    queue = deque()
    queue.append((0,0))   # initial state (jug1, jug2)

    while queue:
        x, y = queue.popleft()

        if (x, y) in visited:
            continue
        visited.add((x, y))

        print("State:", x, y)

        if x == target:
            print("Target achieved!")
            return

        # Production Rules

        # Fill jug1
        queue.append((cap1, y))

        # Fill jug2
        queue.append((x, cap2))

        # Empty jug1
        queue.append((0, y))

        # Empty jug2
        queue.append((x, 0))

        # Pour jug1 -> jug2
        pour = min(x, cap2 - y)
        queue.append((x - pour, y + pour))

        # Pour jug2 -> jug1
        pour = min(y, cap1 - x)
        queue.append((x + pour, y - pour))

water_jug()
